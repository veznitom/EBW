90 double ExpresionCell::RecursiveSolve(Node *node) {
91  //If node has sing its not value node
92  if (node->sign == 0)
93  return node->number;
94  else
95  //If it does it calls RecursiveSolve againg to get values from nodes its connected to
96  switch (node->sign) {
97  case '+':
98  return RecursiveSolve(node->left) + RecursiveSolve(node->right);
99  case '-':
100  return RecursiveSolve(node->left) - RecursiveSolve(node->right);
101  case '*':
102  return RecursiveSolve(node->left) * RecursiveSolve(node->right);
103  case '/':
104  return RecursiveSolve(node->left) / RecursiveSolve(node->right);
105  default:
106  return 0;
107  }
108 }

119 void ExpresionCell::GetValues(const vector<double> &input) {
120  std::ostringstream os;
121  //If the are no values skip calculation
122  if (input.empty()) {
123  number = 0;
124  display = "EMPTY";
125  return;
126  }
127  //If cycle was detected by Table cell is disabled
128  if (killSwitch) {
129  display = "CYCLE";
130  return;
131  }
132  vector<char> backup;
133  vector<Node *> nodes;
134  //For each values in input create value Node
135  for (auto &a:input) {
136  Node *node = new Node;
137  node->sign = 0;
138  node->number = (float) a;
139  node->left = nullptr;
140  node->right = nullptr;
141  nodes.push_back(node);
142  }
143  //Backing up signs so it can be used later again
144  backup = signs;
145  //Creating tree by taking * and / and adjacent value Nodes
146  for (int i = 0; i < (int) signs.size(); ++i) {
147  if (signs[i] == '*' || signs[i] == '/') {
148  Node *connect = new Node;
149  connect->sign = signs[i];
150  connect->left = nodes.at(i);
151  connect->right = nodes.at(i + 1);
152  nodes.erase(nodes.begin() + i);
153  nodes.erase(nodes.begin() + i);
154  nodes.insert(nodes.begin() + i, connect);
155  signs.erase(signs.begin() + i);
156  }
157  }
158  //Creating rest of the nodes from + and -
159  while (nodes.size() > 1) {
160  Node *connect = new Node;
161  connect->sign = signs[0];
162  connect->left = nodes.at(0);
163  connect->right = nodes.at(1);
164  nodes.erase(nodes.begin());
165  nodes.erase(nodes.begin());
166  nodes.insert(nodes.begin(), connect);
167  signs.erase(signs.begin());
168  }
169  //Calling solve method
170  number = RecursiveSolve(nodes.begin().operator*());
171  //Freeing memory
172  RecursiveDelete(nodes.begin().operator*());
173  signs = backup;
174  os << std::setprecision(10) << number;
175  display = os.str();
176 }

178 bool ExpresionCell::Update() {
179  //Removing spaces from text
180  original.erase(remove(original.begin(), original.end(), ' '), original.end());
181  string addr;
182  for (size_t i = 0; i <= original.length(); i++) {
183  //If char in input is +,-,*,/ it's added to sign vector
184  if (original[i] == '+' || original[i] == '-' || original[i] == '*' || original[i] == '/' || original[i] == '\0') {
185  if (original[i] != '\0')
186  signs.push_back(original[i]);
187  //If in the end of text push last address
188  if (original[i] == '\0') {
189  linked.push_back(addr);
190  return true;
191  }
192  //If sign was detected then i means end of last address so it's inserted
193  linked.push_back(addr);
194  addr = "";
195  continue;
196  }
197  //Controlling that there is no invalid input in address
198  if ((original[i] >= 'A' && original[i] <= 'Z') || (original[i] >= '0' && original[i] <= '9')) {
199  addr += original[i];
200  } else {
201  linked.clear();
202  signs.clear();
203  return false;
204  }
205  }
206  linked.clear();
207  signs.clear();
208  return false;
